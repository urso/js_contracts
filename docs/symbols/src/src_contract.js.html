 13     /**
 14      * ErrorClass of namespace Contracts which is thrown when a type error is
 15      * detected.
 16      *
 17      * @param {String} description input String describing the error for
 18      *                 reporting
 19      * @param {Array}  arg Array of arguments the error is created for
 20      *
 21      * @class
 22      */
 23     this.ContractError = function (description, arg) {
 24         /** @field */
 25         this.description = description;
 26 
 27         /** @field */
 28         this.arg = arg;
 29 
 30         return this;
 31     };
 32 
 33     this.ContractError.prototype.run = function () { 
 34         return this; 
 35     };
 36 
 37     /**
 38      * Contract class holding a contracts type description as String to be
 39      * shown on request or when an error is thrown and a predicate input
 40      * arguments are tested with.
 41      *
 42      * @param {Predicate} test the type testing predicate
 43      * @param {String} description type description
 44      *
 45      * @class
 46      */
 47     this.Contract = function (test, description) {
 48         /** 
 49          * type testing predicate.
 50          * @field 
 51          */
 52         this.test = test;
 53 
 54         /** 
 55          * type description
 56          * @field 
 57          * */
 58         this.description = description;
 59         
 60         /**
 61          * Tests the given arguments and throws error if type mismatch is found
 62          *
 63          * @param arg Array of arguments to test
 64          * @return the Contract element itself or a new ContractError(not
 65          *         thrown) element on error
 66          */
 67         this.run = function (arg) {
 68             return test(arg) ? this : new Contracts.ContractError(description, arg);
 69         };
 70 
 71         return this;
 72     };
 73         
 74     /**
 75      * matches any input value.
 76      *
 77      * @constant
 78      */
 79     this.any = new this.Contract(function () { 
 80             return true; 
 81         }, "any");
 82     
 83     /**
 84      * tests if contract matched a given type.
 85      *
 86      * @param c contract application result
 87      * @return true if contract test was successful
 88      */
 89     this.success = function (c) {
 90         return c instanceof Contracts.Contract;
 91     };
 92     
 93     /**
 94      * tests if contract did not matched a given type.
 95      *
 96      * @param c contract application result
 97      * @return true if contract test was not successful
 98      */
 99     this.failure = function (c) {
100         return c instanceof Contracts.ContractError;
101     };

103     var Contract = this.Contract;
104     function mkCheckT(x) {
105         return new Contract(function (arg) { 
106                 return arg !== null && typeof arg === x; 
107             }, "type " + x);
108     }
109     
110     /**
111      * contract testing if input argument is instance of a given class.
112      * @param {Class} x the class inputs must be instance of
113      * @return new contract
114      */
115     this.inst = function (x) {
116         return new this.Contract(function (arg) { 
117                 return arg instanceof x; 
118             }, "instance of " + x.name);
119     };

121     function description_join(cs, word) {
122         if (!cs || cs.length === 0) {
123             return "";
124         }
125         
126         var msg = cs[0].description;
127         for (var i = 1; i < cs.length; i++) {
128             msg += word + cs[i].description;
129         }
130         return msg;
131     }
132     
133     function mk_recorddescription(R) {
134         var msg = "{";
135         var fields = [];
136         for (var f in R) {
137             fields.push(f);
138         }
139         msg += fields[0] + ": " + R[fields[0]].description;
140         for (var i = 1; i < fields.length; i++) {
141             msg += ",\n " + fields[i] + ": " + R[fields[i]].description;
142         }
143         return msg + "\n}";
144     }
145     
146     /**
147      * is the conjunctive contract combinator.
148      *
149      * @param tsts... list of contracts
150      * @return new conjunctive contract (all input contracts must success)
151      */
152     this.and = function (/* tsts */) {
153         var tsts = arguments;
154         function test(arg) {
155             var t;
156             for (var i = 0; i < tsts.length; i++) {
157                 t = tsts[i].run(arg);
158                 if (Contracts.failure(t)) {
159                     return false;
160                 }
161             }
162             return true;
163         }
164         
165         return new Contract(test, "(" + description_join(tsts, ") & (") + ")");
166     };
167     
168     /**
169      * is the disjunctive contract combinator.
170      *
171      * @param tsts... list of contracts
172      * @return new disjunctive contract (just one of the given contracts must
173      *                                   match successfully)
174      */
175     this.or = function (/* tsts */) {
176         var tsts = arguments;
177         function test(arg) {
178             var t;
179             for (var i = 0; i < tsts.length; i++) {
180                 t = tsts[i].run(arg);
181                 if (Contracts.success(t)) {
182                     return true;
183                 }
184             }
185             return false;
186         }
187         
188         return new Contracts.Contract(test, "(" + description_join(tsts, ") | (") + ")");
189     };
190     
191     /**
192      * inverts success and failure of a contract.
193      *
194      * @param tst a contract to create the complement for
195      * @return new complement contract of input contract
196      */
197     this.not = function (tst) {
198         function test(arg) {
199             var t = tst.run(arg);
200             return Contracts.failure(t);
201         }
202         return new Contract(test, "not(" + tst.description + ")");
203     };
204     
205     /**
206      * creates contract from given one which will match null successfully.
207      *
208      * @param tst input contract for case input is not null @return new
209      * contract using input contract if input value is not null
210      */
211     this.maybe = function (tst) {
212         function test(arg) {
213             return arg === null || 
214                    Contracts.success(tst.run(arg));
215         }
216         return new Contracts.Contract(test, "maybe(" + tst.description + ")");
217     };
218 
219     /**
220      * ad hoc polymorphic array contract.
221      *
222      * @param t type every array element must have
223      * @return new contract testing each element in a given array with t.
224      */
225     this.array = function (t) {
226         var any = this.any;
227         var tst = t || any;
228         var test;
229         if (tst === any) {
230             test = function (arg) { 
231                 return arg instanceof Array; 
232             };
233         } else {
234             test = function (arg) {
235                 if (!(arg instanceof Array)) {
236                     return false;
237                 }
238                     
239                 var t;
240                 for (var i = 0; i < arg.length; i++) {
241                     t = tst.run(arg[i]);
242                     if (Contracts.failure(t))  {
243                         return false;
244                     }
245                 }
246                 return true;
247             };
248         }
249             
250         return new Contract(test, "[" + tst.description + "]");
251     };
252         
253     /**
254      * creates a product type contract.<br/>
255      * for example seq(string,number) creates a 
256      * contract for arguments list [string,number].
257      *
258      * @param contracts... input contracts to build product type contract from
259      * @return new product type contract.
260      */
261     this.seq = function () {
262         var cs = arguments.length === 1 && arguments[0] instanceof Array ?
263         arguments[0] : arguments;
264             
265         if (cs.length === 1) {
266             return cs[0];
267         }
268     
269         function test(arg) {
270             if (cs.length !== arg.length) {
271                 return false;
272             }
273             for (var i = 0; i < arg.length; i++) {
274                 if (Contracts.failure(cs[i].run(arg[i]))) {
275                     return false;
276                 }
277             }
278             return true;
279         }
280             
281         return new Contracts.Contract(test, "(" + description_join(cs, ") x (") + ")");
282     };
283         
284     /**
285      * create a contract testing function by building a product type contract
286      * from given contracts.
287      *
288      * @param contracts... input contracts to build product type contract from
289      * @return testing function wich throws ContractError if given arguments
290      *         don't match given contracts else true.
291      */
292     this.contract = function () {
293         var Cs = Contracts.seq.apply(this, arguments);
294         return function (arg) {
295             var t = Cs.run(arg);
296             if (Contracts.success(t)) {
297                 return true;
298             }
299             throw t;
300         };
301     };
302         
303     /**
304      * create a contract testing function by building a product type contract
305      * from given contracts
306      *
307      * @param contract... input contract to build a product type contract from
308      * @return testing function which will return true if given arguments
309      *         match and false else.
310      */
311     this.test = function () {
312         var Cs = Contracts.seq.apply(this, arguments);
313         return function () {
314             var arg = arguments[0].length === 1 ? arguments[0][0] : arguments[0];
315             return Contracts.success(Cs.run(arg));
316         };
317     };
318         
319     /**
320      * record creates a contract from a given Object with each field in the
321      * object having a type description. When testing the contract the input
322      * is tested for each field in the type pattern object. If the testing
323      * object has a missing field or a type mismatch in any field is found the
324      * contract will fail.
325      *
326      * @param {Object} R object with field type description.
327      * @return a new contract testing the structure of its values
328      */
329     this.record = function (R) {
330         function test(arg) {
331             if (!(arg instanceof Object)) {
332                 return arg;
333             }
334                 
335             for (var field in arg) {
336                 if (arg[field] === undefined || 
337                     Contracts.failure(R[field].run(arg[field]))) 
338                 {
339                     return false;
340                 }
341             }
342             return true;
343         }
344         return new Contracts.Contract(test, mk_recorddescription(R));
345     };

347     /**
348      * Class for describing function/method contracts with input and output
349      * contracts.
350      *
351      * DON'T USE THIS CLASS DIRECTLY. USE FUNCTION "proc" INSTEAD.
352      *
353      * @class
354      * @private
355      */
356     this.ProcContract = function (input, ret) {
357         this.input = input;
358         this.ret = ret;
359         this.description = input.description + " -> " + ret.description;
360         return this;
361     };

363     /**
364      * Describe the signature of a function. The last function argument is used to
365      * be the functions return value type.
366      *
367      * @param types... list of type descriptions. Last one is the function's
368      *                 return type.
369      *
370      * @return type description for functions
371      */
372     this.proc = function () {
373         var input = new Array(arguments.length);
374         for (var i = 0; i < arguments.length; i++) {
375             input[i] = arguments[i];
376         }
377         var ret = input.pop();
378         
379         return new Contracts.ProcContract(Contracts.seq(input), ret);
380     };

432     function mk_testFnContract(signature, symb) {
433         return function (proceed, args) {
434             if (!Contracts.test(signature.input)(args)) {
435                 throw new Error("input type mismatch in " + symb + 
436                                 ": " + signature.input.description);
437             }
438             
439             var ret = proceed.apply(this, args);
440             if (signature.ret instanceof Contracts.ProcContract) {
441                 if (typeof ret !== 'function') {
442                     throw new Error("output type mismatch in " + symb + 
443                                     ": " + signature.ret.description +
444                                     "\nbut returned value is no function: " + ret);
445                 } else if (!ret.$sigFn) { //add signature contract if function has non yet
446                     var retTstFn = mk_testFnContract(signature.ret, symb + "->anonym");
447                     var fn = function () {
448                         return retTstFn(ret, arguments);
449                     };
450                     fn.$sigFn = true;
451                     return fn;
452                 }
453             } else if (!Contracts.test(signature.ret)(ret)) {
454                 throw new Error("output type mismatch in " + symb + 
455                                 ": " + signature.ret.description +
456                                 "\nbut returned" + ret);
457             }
458             
459             return ret;
460         };
461     }

463     /**
464      * Adds a Signature to a given Object/Namespace/Class.  The signature
465      * itself is an object with every field having a type description for each
466      * method/field to annotate in the input object.  Fields
467      * (variables/constants) are tested with the given contract and methods
468      * are overwritten using AOP to do type checking first when called.
469      *
470      * @param {Object} sig signature to add to object
471      * @param {Object} obj the object to annotate
472      *
473      * @throws Error if a symbol in sig is not found in obj or a field/constant
474      *               in object has a wrong type.
475      */
476     this.addSignature = function (sig, obj) {
477         Contracts.contract(Contracts.object, Contracts.object)(arguments);
478     
479         for (var symb in sig) {
480             if (obj[symb] === 'undefined') {
481                 throw new Error("signature symbol " + symb + "not found in object");
482             }
483         
484             //check if signature and objects symbol
485             //type match
486             
487             if (typeof obj[symb] === 'function' && sig[symb] instanceof Contracts.ProcContract) {
488                 var fn = mk_testFnContract(sig[symb], symb);
489                 fn.$sigFn = true;
490                 
491                 AOP.around(obj, symb, fn);
492             } else if (!Contracts.test(sig[symb])(obj[symb])) {
493                 throw new Error("Signature type mismatch for " + symb);
494             }
495         }
496     };
